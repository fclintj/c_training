 64 user_loan calculate_loan_info(struct user_loan loan){
 65     double i = (loan.interest_rate/100.0)/12.0;
 66     int n = loan.years * 12;
 67     double discount = (pow(1+i,n)-1)/(i*pow(1+i,n));
 68     loan.monthly_payment = loan.loan_amount / discount;
 69     loan.loan_payment = loan.monthly_payment * n;
 70     return loan;
 71 }

 78 void calculate_quardinate_value(struct degree_coordinate *input){
 79     if(input->circular_vaule == 90) {snprintf(input->quardinate_vaule, char_buffer_size,"E");}
 80     else if(input->circular_vaule == 180) {snprintf(input->quardinate_vaule, char_buffer_size,"S");}
 81     else if(input->circular_vaule == 270) {snprintf(input->quardinate_vaule, char_buffer_size,"W");}
 82     else if(input->circular_vaule == 0 || input->circular_vaule == 0) {
 83         snprintf(input->quardinate_vaule, char_buffer_size,"N");
 84     }
 85 
 86 
 87     else if(input->circular_vaule > 0 &&  input->circular_vaule < 90) {
 88         snprintf(input->quardinate_vaule, char_buffer_size,"N %.2lf° E",input->circular_vaule);
 89     }
 90 
 91     else if(input->circular_vaule > 90 &&  input->circular_vaule < 180) {
 92         snprintf(input->quardinate_vaule, char_buffer_size,"S %.2lf° E",180 - input->circular_vaule);
 93     }
 94 
 95     else if(input->circular_vaule > 180 &&  input->circular_vaule < 270) {
 96         snprintf(input->quardinate_vaule, char_buffer_size,"S %.2lf° W",input->circular_vaule - 180);
 97     }
 98 
 99     else if(input->circular_vaule > 270 &&  input->circular_vaule < 360) {
100         snprintf(input->quardinate_vaule, char_buffer_size,"N %.2lf° W", 360 - input->circular_vaule);
101     }
102 
103     else {
104         printf("You must enter a value between 0 - 360°");
105         exit(EXIT_FAILURE);
106     }
107 }
